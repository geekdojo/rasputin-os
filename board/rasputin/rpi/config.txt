# Raspberry Pi firmware config — rpi SKU (arm64: Pi 4 / Pi 5 / CM5).
# Installed into the firmware output dir via BR2_PACKAGE_RPI_FIRMWARE_CONFIG_FILE
# and staged onto the boot FAT by board/rasputin/common/post-image.sh.
#
# One unified image, TWO kernels: the firmware auto-selects the matching DTB by
# board, and the [pi4]/[pi5] sections below pick the matching kernel — so the
# same FAT boots a Pi 4 (BCM2711) or a Pi 5 / CM5 (BCM2712). config.txt is read
# from the boot partition on every model.

# 64-bit. The kernel is chosen per board in the [pi4]/[pi5] sections below
# (kernel8.img = bcm2711/Pi 4, kernel_2712.img = bcm2712/Pi 5) — NOT a global
# `kernel=`, since the two SoCs need different kernels.
arm_64bit=1
auto_initramfs=1
disable_overscan=1

# Serial console on the GPIO UART (GPIO 14/15 → ttyAMA0). REQUIRED for bring-up:
# with this on, the GPU firmware prints its boot banner to the UART before the
# kernel even starts, so a USB-UART gives us a window into the firmware→kernel
# handoff even if the kernel console node turns out to differ on the Pi 5 (RP1
# enumerates UARTs differently — VERIFY the kernel console= on first boot).
enable_uart=1

# Per-model knobs (firmware applies the matching [block]).
[pi4]
# Pi 4 (bcm2711): its GPU firmware (start4.elf/fixup4.dat) loads this kernel.
kernel=kernel8.img
[pi5]
# Pi 5 / CM5 (bcm2712): the SPI-EEPROM bootloader loads this kernel via config.txt.
kernel=kernel_2712.img
# Enable the external PCIe x1 lane so the kernel sees an NVMe SSD on the M.2 /
# SSD HAT (Pi 5 PCIe is OFF by default). Required for booting/running from NVMe;
# harmless when no PCIe device is attached (just powers the controller). The
# EEPROM bootloader has its own NVMe support for finding the boot partition;
# this is what lets the OS use the drive once booted.
dtparam=pciex1
[all]
