# Raspberry Pi A/B tryboot selector — read by the firmware BEFORE config.txt.
# Staged onto the boot FAT root by board/rasputin/common/post-image.sh.
#
# tryboot_a_b=1 makes a one-shot `tryboot` reboot re-read the NORMAL config.txt
# (not tryboot.txt), so our config.txt's [tryboot] section can swap the kernel
# cmdline to the candidate rootfs slot. We deliberately do NOT switch
# boot_partition here: the image has a SINGLE boot FAT shared by both slots
# (one kernel set + firmware), and only the rootfs is A/B — exactly like the
# n100 (one /bzImage, A/B squashfs). The slot is chosen entirely by which
# cmdline the firmware loads: cmdline.txt = the committed slot, tryboot.txt =
# the trial slot. A failed trial (no mark-good) consumes the one-shot and the
# next normal boot reverts to cmdline.txt — that IS the rollback.
#
# Driven by the RAUC custom backend at /usr/lib/rauc/rpi-tryboot-backend.sh.
# See os-images/buildroot-os.md §3.
[all]
tryboot_a_b=1
